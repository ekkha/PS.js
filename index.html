            var Minimal_Angle = (h, m) => 
            {
               let hourAngle=(30*h)+(0.5*m);
               let minuteAngle=6*m;
               let angle=Math.abs(hourAngle-minuteAngle);
               if(angle>180)
               {
                 angle=360-angle;
               }
               return angle;
            };

Q4:Reverse a Number.?

Ans:
             
var Reverse_Number = (N) => 
{
  
  let ans=0;
  let rem=0;
  while(N>0){
    rem=N%10;
    ans=ans*10+rem;
    N=Math.floor(N/10);
  }
  return ans;
 

};
